Why Is This an Issue?

CWE-90 flags code that constructs LDAP queries using unsanitized user input. When user-controlled data is concatenated directly into an LDAP search filter, an attacker can inject LDAP filter metacharacters — (, ), *, \, |, & — to modify the query's logic.

Successful LDAP injection can allow an attacker to:

  • Bypass authentication — inject a filter that always evaluates to true (e.g., *)(uid=*))(|(uid=*)
  • Enumerate users and groups — extract directory information by manipulating search filters
  • Elevate privileges — modify filters to return admin accounts or privileged group memberships
  • Exfiltrate data — read sensitive attributes (emails, phone numbers, organizational structure) from the directory

This weakness maps to OWASP A03:2021 — Injection. LDAP injection is especially dangerous in enterprise environments where Active Directory or OpenLDAP backs authentication and authorization. The fix requires escaping LDAP special characters in user input or using framework-provided filter builders that handle encoding automatically.

Examples of the Issue

Dangerous: String concatenation in JNDI LDAP filters (Java)

Directly concatenating user input into LDAP filter strings via DirContext.search() is the classic injection vector. An attacker submitting *)(uid=*))(|(uid=* as a username transforms the filter into one that matches all users, bypassing authentication.

// DANGEROUS: User input concatenated directly into LDAP filter. import javax.naming.directory.*; public class LdapAuthService { public boolean authenticate(String username, String password) { DirContext ctx = new InitialDirContext(env); // ❌ Attacker input: username = "*)(uid=*))(|(uid=*" // Resulting filter: (&(uid=*)(uid=*))(|(uid=*)(userPassword=...)) // This matches ALL users — authentication bypassed. String filter = "(&(uid=" + username + ")" + "(userPassword=" + password + "))"; NamingEnumeration<?> results = ctx.search( "ou=users,dc=example,dc=com", filter, // ❌ Unescaped user input in filter! new SearchControls() ); return results.hasMore(); } } // Also dangerous — in a user lookup service: public UserInfo findUser(String searchTerm) { // ❌ searchTerm = "*)(objectClass=*" → returns all objects. String filter = "(|(cn=" + searchTerm + ")" + "(mail=" + searchTerm + "))"; return ldapSearch(filter); }

Dangerous: Raw filter strings in Spring LDAP and Spring Security (Java/Spring)

Using HardcodedFilter or raw string filters in Spring's LdapTemplate bypasses Spring LDAP's built-in encoding. Similarly, custom Spring Security LDAP configurations that build search filters via string concatenation are vulnerable even though the framework provides safe alternatives.

// DANGEROUS: Raw string filter in Spring LdapTemplate. import org.springframework.ldap.core.LdapTemplate; import org.springframework.ldap.filter.HardcodedFilter; @Service public class LdapUserService { @Autowired private LdapTemplate ldapTemplate; public List<User> searchUsers(String query) { // ❌ HardcodedFilter passes the string as-is — no encoding. String filter = "(|(cn=" + query + ")(sn=" + query + "))"; return ldapTemplate.search( "ou=users", filter, // ❌ Raw, unescaped filter! new UserAttributesMapper() ); } // Also dangerous — using LdapTemplate with string format: public User findByEmail(String email) { String filter = String.format("(mail=%s)", email); // ❌ email = "admin@co)(uid=*" → matches unintended entries. return ldapTemplate.searchForObject("ou=users", filter, mapper); } } // Also dangerous — MessageFormat in LDAP auth provider: // spring.security.ldap.user-search-filter=(uid={0}) // If overridden with raw string building, injection is possible.

Dangerous: Template literals in ldapjs search filters (Node.js)

Node.js LDAP libraries like ldapjs and ldapts accept string filters. Using template literals or concatenation to insert user input creates the same injection risk. The ldapjs library provides filter classes but many codebases skip them for convenience.

// DANGEROUS: Template literal in ldapjs filter — LDAP injection. const ldap = require('ldapjs'); const client = ldap.createClient({ url: 'ldap://ldap.example.com' }); function lookupUser(username) { return new Promise((resolve, reject) => { // ❌ username = "*)(objectClass=*" → matches everything. const filter = `(uid=${username})`; client.search('ou=users,dc=example,dc=com', { filter: filter, // ❌ Unescaped user input! scope: 'sub' }, (err, res) => { const entries = []; res.on('searchEntry', e => entries.push(e.ppiObject)); res.on('end', () => resolve(entries)); }); }); } // Also dangerous — in Express route with concatenation: app.get('/api/users/search', (req, res) => { const q = req.query.q; // ❌ q = "*)(cn=admin)" → retrieves admin entries. const filter = `(|(cn=${q})(mail=${q}))`; ldapSearch(filter).then(users => res.json(users)); });

How to Resolve

Fix (Java): Escape LDAP metacharacters or use filter parameterization

For raw JNDI, escape all LDAP special characters (* ( ) \ / NUL) in user input before inserting into filter strings. Better yet, use JNDI's built-in parameterized filter format with filterArgs, which handles escaping automatically.

// SAFE: JNDI parameterized filter — automatic escaping. import javax.naming.directory.*; public class LdapAuthService { public boolean authenticate(String username, String password) { DirContext ctx = new InitialDirContext(env); // ✅ Parameterized filter — {0} and {1} are auto-escaped. String filter = "(&(uid={0})(userPassword={1}))"; Object[] filterArgs = { username, password }; NamingEnumeration<?> results = ctx.search( "ou=users,dc=example,dc=com", filter, filterArgs, // ✅ Special chars escaped automatically. new SearchControls() ); return results.hasMore(); } // Manual escaping helper (RFC 4515): public static String escapeLdapFilter(String input) { if (input == null) return null; StringBuilder sb = new StringBuilder(); for (char c : input.toCharArray()) { switch (c) { case '*': sb.append("\\2a"); break; case '(': sb.append("\\28"); break; case ')': sb.append("\\29"); break; case '\\': sb.append("\\5c"); break; case '\0': sb.append("\\00"); break; default: sb.append(c); } } return sb.toString(); } }

Fix (Spring): Use Spring LDAP's type-safe filter builder + LdapQueryBuilder

Spring LDAP provides LdapQueryBuilder and filter classes (EqualsFilter, OrFilter, etc.) that automatically encode LDAP special characters. For Spring Security LDAP, use the built-in {0} placeholder in search filters, which is auto-escaped by the framework.

// SAFE: Spring LDAP type-safe filter builder — auto-encoding. import org.springframework.ldap.core.LdapTemplate; import org.springframework.ldap.query.LdapQueryBuilder; import org.springframework.ldap.filter.*; import static org.springframework.ldap.query.LdapQueryBuilder.query; @Service public class LdapUserService { @Autowired private LdapTemplate ldapTemplate; // ─── Pattern 1: LdapQueryBuilder (preferred) ─── public List<User> searchUsers(String searchTerm) { return ldapTemplate.search( query() .base("ou=users") .where("cn").is(searchTerm) // ✅ Auto-escaped. .or("sn").is(searchTerm) // ✅ Auto-escaped. .or("mail").is(searchTerm), // ✅ Auto-escaped. new UserAttributesMapper() ); } // ─── Pattern 2: Filter objects ─── public User findByEmail(String email) { Filter filter = new EqualsFilter("mail", email); // ✅ Auto-escaped. return ldapTemplate.searchForObject( "ou=users", filter.encode(), mapper ); } // ─── Pattern 3: OrFilter for multi-attribute search ─── public List<User> multiSearch(String term) { OrFilter orFilter = new OrFilter(); orFilter.or(new LikeFilter("cn", term)); // ✅ Wildcards escaped. orFilter.or(new LikeFilter("mail", term)); return ldapTemplate.search("ou=users", orFilter.encode(), mapper); } } // Spring Security LDAP — {0} placeholder is auto-escaped: // auth.ldapAuthentication() // .userSearchFilter("(uid={0})") // ✅ Safe — {0} is encoded. // .groupSearchFilter("(member={0})")

Fix (JS): Use ldapjs filter objects or manual escaping (Node.js)

Use ldapjs filter classes (EqualityFilter, OrFilter) instead of string templates. These classes encode special characters automatically. If you must use string filters, apply manual escaping per RFC 4515 before interpolation.

// SAFE: ldapjs filter objects — automatic encoding. const ldap = require('ldapjs'); function lookupUser(username) { // ✅ EqualityFilter encodes LDAP metacharacters automatically. const filter = new ldap.EqualityFilter({ attribute: 'uid', value: username // * ( ) \ are escaped by the filter class. }); return new Promise((resolve, reject) => { client.search('ou=users,dc=example,dc=com', { filter: filter, // ✅ Safe — uses filter object. scope: 'sub' }, (err, res) => { const entries = []; res.on('searchEntry', e => entries.push(e.ppiObject)); res.on('end', () => resolve(entries)); }); }); } // Multi-attribute search with OrFilter: function searchUsers(term) { const filter = new ldap.OrFilter({ filters: [ new ldap.EqualityFilter({ attribute: 'cn', value: term }), new ldap.EqualityFilter({ attribute: 'mail', value: term }) ] }); return ldapSearch(filter); } // Manual escaping function (RFC 4515) if filter objects aren't available: function escapeLdapFilter(input) { return input.replace(/([\\*()\/\0])/g, (match) => { return '\\' + match.charCodeAt(0).toString(16).padStart(2, '0'); }); } // Usage: const safe = `(uid=${escapeLdapFilter(username)})`;
